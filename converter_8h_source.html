190 {
191 static_assert(MatrixTraits<m_in, T>::size == DIM_4D && MatrixTraits<m_out, T>::size == DIM_4D);
192

202 const auto& asgn_x = ttt[x];
203 T val = MatrixTraits<m_in, T>::get(in, y, x);
